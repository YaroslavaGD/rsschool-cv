function bouncingBall(h, bounce, window) {
  let numDrops = 0
  let currentHeight = h;

  if (bounce >= 1 ||  bounce <= 0) return -1;

  while (currentHeight > window) {
      currentHeight *= bounce;
      numDrops = numDrops + 2;
  }

  return numDrops - 1;
}
